177 QTextBrowser* constructDijkstra(const QString START_VERTEX)
178{
179 QTextBrowser* output = new QTextBrowser;
180 const int START_INDEX = getIndexFromValue(START_VERTEX);
181
182 // Create a priority queue to store vertices that
183 // are being preprocessed. This is weird syntax in C++.
184 priority_queue<iPair, vector<iPair>, greater<iPair> > pq;
185 // Container for every vertice traversed from src to destination
186 vector<int> vertices_to_destination;
187
188 // Create a vector for distances and initialize all
189 // distances as infinite (INF)
190 vector<int> dist(vertex_name_array.size(), INF);
191
192 // Insert source itself in priority queue and initialize
193 // its distance as 0.
194 pq.push(make_pair(0, START_INDEX));
195 dist[START_INDEX] = 0;
196 vertices_to_destination.push_back(START_INDEX);
197
198 /* Looping till priority queue becomes empty (or all
199 distances are not finalized) */
200 while (!pq.empty()) {
201
202 // The first vertex in pair is the minimum distance
203 // vertex, extract it from priority queue.
204 // vertex label is stored in second of pair (it
205 // has to be done this way to keep the vertices
206 // sorted distance (distance must be first item
207 // in pair)
208 int u = pq.top().second;
209 pq.pop();
210
211 // 'i' is used to get all adjacent vertices of a
212 // vertex
213 list<pair<int, int> >::iterator i;
214 int save_v = 0;
215 for (i = weight_matrix[u].begin(); i != weight_matrix[u].end(); ++i) {
216 // Get vertex label and weight of current
217 // adjacent of u.
218 int v = (*i).first;
219 int weight = (*i).second;
220
221 // If there is a shorter path to v through u.
222 if (dist[v] > dist[u] + weight) {
223 // Updating distance of v
224 dist[v] = dist[u] + weight;
225 pq.push(make_pair(dist[v], v));
226
227 save_v = v;
228 }
229 }
230 vertices_to_destination.push_back(save_v);
231
232 }
233
234 // Print shortest distances stored in dist[]
235 output->append( "\n\nCity Distance from " + vertex_name_array[START_INDEX] + " to\n");
236 for (int i = 0; i < vertex_name_array.size(); ++i)
237 {
238 output->append( "\t" + vertex_name_array[i] + ": " + static_cast<QChar>(dist[i]) + " km\n");
239 }
